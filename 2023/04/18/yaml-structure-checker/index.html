    def self.nested_keys_to_nested_keys_each_env(nested_keys, envs)
      nested_keys_each_env = {}
      envs.each do |env|
        nested_keys_each_env[env] = []
      end

      nested_keys.each do |nested_key|
        nested_key = nested_key.clone

        end
      end

      nested_keys_each_env
    end
